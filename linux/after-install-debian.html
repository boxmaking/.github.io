1. 校正日期和时间

$cp /usr/share/zoneinfo/Asia/ShangHai  /etc/localtime     #时区为亚洲/上海
  1. 网络校时
$apt-get install ntpdate
$ntpdate 210.72.145.44        # 中国国家时间服务器: 210.72.145.44
  1. 手动校时
$sudo date -s 11/13/2019                #2019年11月13日
$sudo date -s 10:05:30                  #10点05分30秒
